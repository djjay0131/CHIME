\section{Paper Overview}

\begin{frame}{CHIME: Key Idea}
\begin{itemize}
    \item Disaggregated memory (DM): decouple compute and memory into network-connected pools
    \item Range indexes on DM face a trade-off: \textbf{cache consumption} vs.\ \textbf{read amplifications}
    \item CHIME: hybrid B+ tree with \textbf{hopscotch-hashing leaf nodes}
    \begin{itemize}
        \item B+ tree internals $\rightarrow$ low cache consumption
        \item Hopscotch leaves $\rightarrow$ low read amplifications
    \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Three Key Techniques}
\begin{enumerate}
    \item \textbf{Three-level optimistic synchronization}\\
    Cache line versions + reused hopscotch bitmaps for fine-grained concurrency
    \item \textbf{Access-aggregated metadata management}\\
    Vacancy bitmap piggybacking + leaf metadata replication
    \item \textbf{Hotness-aware speculative read}\\
    Hotspot buffer to shortcut neighborhood reads
\end{enumerate}
\end{frame}
